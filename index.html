   优先级队列 priority_queue 头文件 queue
   可以把放进去的元素按照一定的规则进行自动的排序，底层 堆排序
   priority_queue pq;//默认从大到小的
   priority_queue<数据类型, 底层容器, 排序规则> pq;
   底层容器 ：vector<数据类型> / deque<数据类型>
   排序规则 ：
   1.基本数据类型 + string
   (1) less<类型> 默认的 降序的
   priority_queue, less > pq;
   (2) greater<类型> 升序的
   priority_queue, greater > pq;
    2.结构体类型 结构体 node ，成员 x, y
   bool operator <(node a, node b){//priority_queue pq;
            return a.x < b.x;// 按照 x 从大到小
            return a.x > b.x;// 按照 x 从小到大
   }
   struct cmp{//priority_queue, cmp > pq;
        bool operator()(const node &a, const node &b) const{
            return a.x < b.x;// 按照 x 从大到小
            return a.x > b.x;// 按照 x 从小到大
        }
   };
